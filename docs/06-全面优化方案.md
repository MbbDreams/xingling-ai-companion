```python
class SmartCache:
    """多级缓存策略"""
    
    # L1: 内存缓存 (最近对话上下文)
    _memory_cache: Dict[int, ConversationContext] = {}
    
    # L2: Redis 缓存 (用户信息、记忆摘要)
    _redis: Redis
    
    # L3: 数据库 (完整数据)
    
    async def get_conversation_context(self, user_id: int) -> ConversationContext:
        # 1. 检查内存缓存
        if user_id in self._memory_cache:
            return self._memory_cache[user_id]
        
        # 2. 检查 Redis
        cached = await self._redis.get(f"ctx:{user_id}")
        if cached:
            context = ConversationContext.parse_raw(cached)
            self._memory_cache[user_id] = context
            return context
        
        # 3. 从数据库加载
        context = await self._load_from_db(user_id)
        
        # 4. 写入缓存
        await self._redis.setex(f"ctx:{user_id}", 300, context.json())
        self._memory_cache[user_id] = context
        
        return context
```

#### 3.1 多级缓存架构

```python
class CacheManager:
    """统一缓存管理"""
    
    def __init__(self, redis: Redis):
        self.redis = redis
        self.local_cache = {}  # L1: 本地内存
        self.hit_count = 0
        self.miss_count = 0
    
    async def get(self, key: str, ttl: int = 300) -> Any:
        # L1: 本地缓存
        if key in self.local_cache:
            self.hit_count += 1
            return self.local_cache[key]
        
        # L2: Redis
        value = await self.redis.get(key)
        if value:
            self.hit_count += 1
            parsed = json.loads(value)
            self.local_cache[key] = parsed  # 回填本地缓存
            return parsed
        
        self.miss_count += 1
        return None
    
    async def set(self, key: str, value: Any, ttl: int = 300):
        # 写入两级缓存
        self.local_cache[key] = value
        await self.redis.setex(key, ttl, json.dumps(value))
    
    @property
    def hit_rate(self) -> float:
        total = self.hit_count + self.miss_count
        return self.hit_count / total if total > 0 else 0
```

#### 3.2 限流与熔断

```python
class RateLimiter:
    """基于 Redis 的分布式限流"""
    
    async def check_rate_limit(
        self, 
        key: str,  # user_id 或 ip
        max_requests: int = 100,
        window: int = 60  # 秒
    ) -> bool:
        """滑动窗口限流"""
        current = await self.redis.get(f"ratelimit:{key}")
        if current and int(current) >= max_requests:
            return False
        
        pipe = self.redis.pipeline()
        pipe.incr(f"ratelimit:{key}")
        pipe.expire(f"ratelimit:{key}", window)
        await pipe.execute()
        return True

class CircuitBreaker:
    """熔断器 - 防止级联故障"""
    
    def __init__(self, failure_threshold: int = 5, recovery_timeout: int = 60):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        self.failure_count = 0
        self.last_failure_time = None
    
    async def call(self, func, *args, **kwargs):
        if self.state == "OPEN":
            if time.time() - self.last_failure_time > self.recovery_timeout:
                self.state = "HALF_OPEN"
            else:
                raise CircuitBreakerOpen("服务暂时不可用")
        
        try:
            result = await func(*args, **kwargs)
            if self.state == "HALF_OPEN":
                self.state = "CLOSED"
                self.failure_count = 0
            return result
        except Exception as e:
            self.failure_count += 1
            self.last_failure_time = time.time()
            
            if self.failure_count >= self.failure_threshold:
                self.state = "OPEN"
            
            raise e
```

#### 3.3 分布式锁

```python
class DistributedLock:
    """Redis 分布式锁"""
    
    async def acquire(self, lock_key: str, timeout: int = 10) -> bool:
        """获取锁"""
        identifier = str(uuid.uuid4())
        lock_key = f"lock:{lock_key}"
        
        # SET key value NX EX timeout
        acquired = await self.redis.set(
            lock_key, identifier, nx=True, ex=timeout
        )
        
        if acquired:
            self.identifier = identifier
            return True
        return False
    
    async def release(self, lock_key: str):
        """释放锁（使用 Lua 脚本保证原子性）"""
        lua_script = """
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
        """
        await self.redis.eval(
            lua_script, 1, f"lock:{lock_key}", self.identifier
        )

# 使用场景：防止重复完成任务
async def complete_task(user_id: int, task_id: int):
    lock_key = f"task:{user_id}:{task_id}"
    
    lock = DistributedLock(redis)
    if not await lock.acquire(lock_key, timeout=5):
        raise HTTPException(429, "操作太频繁")
    
    try:
        # 执行业务逻辑
        await _do_complete_task(user_id, task_id)
    finally:
        await lock.release(lock_key)
```

```python
class ConversationManager:
    """对话状态管理"""
    
    def __init__(self):
        self.sessions: Dict[str, ConversationSession] = {}
    
    async def get_or_create_session(self, user_id: int) -> ConversationSession:
        """获取或创建对话会话"""
        session_key = f"session:{user_id}"
        
        if session_key not in self.sessions:
            # 从 Redis 恢复或创建新会话
            session_data = await redis.get(session_key)
            if session_data:
                self.sessions[session_key] = ConversationSession.parse_raw(session_data)
            else:
                self.sessions[session_key] = ConversationSession(user_id=user_id)
        
        return self.sessions[session_key]
    
    async def update_session(self, user_id: int, message: Message, response: str):
        """更新会话状态"""
        session = await self.get_or_create_session(user_id)
        
        # 更新对话历史
        session.add_turn(message, response)
        
        # 检测话题转换
        if session.detect_topic_shift():
            session.archive_current_topic()
        
        # 保存到 Redis
        await redis.setex(f"session:{user_id}", 3600, session.json())
```

```python
class ContentSafety:
    """内容安全检查"""
    
    async def check_message(self, content: str) -> SafetyResult:
        """检查用户消息"""
        # 1. 敏感词检测
        if self.contains_sensitive_words(content):
            return SafetyResult(blocked=True, reason="敏感内容")

                blocked=False,

        return SafetyResult(blocked=False)
```
